! ============================================================
!  Lab 01. Routed Port + EIGRP  (Preconfig)
!  Topology : SW1 - SW2 - SW3 / SW2 - R4 - SW4
!  Routing  : EIGRP AS 100 (no auto-summary)
! ============================================================
!
!
! ############################################################
! #  SW1
! ############################################################
en
conf t
!
! ===== STEP 1. ip routing =====
ip routing
!
! ===== STEP 2. Loopback =====
interface loopback 0
 ip address 13.13.10.1 255.255.255.0
!
! ===== STEP 3. Routed Port (no switchport) =====
int e0/0
 no switchport
 ip address 13.13.12.1 255.255.255.0
 no sh
!
! ===== STEP 4. EIGRP 100 =====
router eigrp 100
 no auto-summary
 network 13.13.10.0 0.0.0.255
 network 13.13.12.0 0.0.0.255
!
end
!
!
